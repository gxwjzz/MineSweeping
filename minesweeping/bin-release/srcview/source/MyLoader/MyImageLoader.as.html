package MyLoader
{
    import flash.display.Bitmap;
    import flash.display.Loader;
    import flash.display.Sprite;
    import flash.events.*;
    import flash.net.URLRequest;


    public class MyImageLoader extends Sprite
    {
        public var images:Array;
        private var size = 0;
        
        
        public function MyImageLoader()
        {
            images = new Array();
        }
        
        public function load(urlList:Array):void {
            if(!urlList) 
            {  
                trace("Empty List");
                return; 
            }
            
            size = urlList.length;
        
            for(var i:int=0; i<urlList.length; i++)
                loadImage(urlList[i]);
                
        }

        private function loadImage(url:String):void {
                var loader:Loader = new Loader();
                loader.contentLoaderInfo.addEventListener(Event.COMPLETE, onLoadingComplete);
                loader.contentLoaderInfo.addEventListener(IOErrorEvent.IO_ERROR, ioErrorHandler);    
                var request:URLRequest = new URLRequest(url);
                loader.load(request);
        }

        private function ioErrorHandler(event:IOErrorEvent):void {
                trace("Unable to load image");
           }

        private function onLoadingComplete(event:Event):void {
                var loader:Loader = Loader(event.target.loader);
       
                images.push({url:loader.contentLoaderInfo.url, data:Bitmap(loader.content)});
                if(images.length == size)
                {
                    dispatchEvent(new Event("AllImagesLoaded"));
                }

        }
        
        public function getImageByURL(url:String):Object {
                for(var i:int = 0; i < images.length; i++)
                    if(images[i].url.indexOf(url) >= 0) return images[i].data;                    
                return null; 
        }


    }
}

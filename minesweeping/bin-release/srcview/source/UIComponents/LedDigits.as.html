package UIComponents
{
    import flash.display.Bitmap;
    import flash.display.BitmapData;
    import flash.display.Sprite;
    import flash.geom.Matrix;
    import flash.geom.Point;
    import flash.geom.Rectangle;

    
    public class LedDigits extends Sprite
    {
        private var digitRes:Bitmap, digitsDisplay:Bitmap;
        private const digitWidth:int = 16;
        private const digitHeight:int = 24;
        
        public function LedDigits(resource:Bitmap)
        {
            //scale resource image to a suitable size
               digitRes = new Bitmap();
            digitRes.bitmapData = new BitmapData(digitWidth*10, digitHeight, true, 0x000000);
            var scaleMat:Matrix = new Matrix();
            scaleMat.scale(digitWidth*10/resource.width, digitHeight/resource.height);
            digitRes.bitmapData.draw(resource, scaleMat, null, "add"); 
            //create display image
            digitsDisplay = new Bitmap();
            digitsDisplay.bitmapData = new BitmapData(digitWidth*2 - 3, digitHeight, false, 0x000000);
            addChild(digitsDisplay);
            drawNumber(99);            
        }
        
        public function drawNumber(num:int):void {
            //calculate tens and ones
            var tens:int = 0, ones:int = 0;
            if(num > 99) num = 99;
            if(num < 0) num = 0;
            tens = int(num / 10);
            ones = num % 10;
            
            digitsDisplay.bitmapData.copyPixels(digitRes.bitmapData, new Rectangle(tens * digitWidth, 0 ,digitWidth, digitHeight), new Point(0,0));                    
            digitsDisplay.bitmapData.copyPixels(digitRes.bitmapData, new Rectangle(ones * digitWidth, 0 ,digitWidth, digitHeight), new Point(digitWidth,0));
        }


    }
}

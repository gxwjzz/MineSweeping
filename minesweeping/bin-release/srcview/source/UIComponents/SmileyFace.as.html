package UIComponents
{
    import flash.display.Bitmap;
    import flash.display.BitmapData;
    import flash.display.Sprite;
    import flash.geom.Matrix;
    import flash.geom.Point;
    import flash.geom.Rectangle;
    
    public class SmileyFace extends Sprite
    {
        private var smile:Bitmap, cool:Bitmap, wrong:Bitmap, faceDisplay:Bitmap;
        private const imgSize:int = 30;
        private const resSize:int = 80;
        private var state:int = 0; // 0 - smile, 1 - cool, 2 - sad
        
        public function SmileyFace(smile_res:Bitmap, cool_res:Bitmap, wrong_res:Bitmap)
        {
            //scale resource image to a suitable size
            var scaleMat:Matrix = new Matrix();
               smile = new Bitmap();
            smile.bitmapData = new BitmapData(imgSize, imgSize, true);
            scaleMat.identity();
            scaleMat.scale(imgSize/smile_res.width, imgSize/smile_res.height);
            smile.bitmapData.draw(smile_res, scaleMat, null, null, null, true); 

               cool = new Bitmap();
            cool.bitmapData = new BitmapData(imgSize, imgSize, true);
            scaleMat.identity();
            scaleMat.scale(imgSize/cool_res.width, imgSize/cool_res.height);
            cool.bitmapData.draw(cool_res, scaleMat, null, null, null, true); 

               wrong = new Bitmap();
            wrong.bitmapData = new BitmapData(imgSize, imgSize, true);
            scaleMat.identity();
            scaleMat.scale(imgSize/wrong_res.width, imgSize/wrong_res.height);
            wrong.bitmapData.draw(wrong_res, scaleMat, null, null, null, true); 

            //create display image
            faceDisplay = new Bitmap();
            faceDisplay.bitmapData = new BitmapData(imgSize, imgSize, false, 0xFFFFFF);
            addChild(faceDisplay);
            setFace(0);
        }
        
        public function setFace(_state:int):void {
            this.state = _state;
            var faceRes:Bitmap;
            switch(state) {
                case 0: faceRes = smile;
                    break;
                case 1: faceRes = cool;
                    break;
                case 2: faceRes = wrong;
                    break;
                default: return;
            }
            faceDisplay.bitmapData.copyPixels(faceRes.bitmapData, new Rectangle(0, 0, imgSize, imgSize), new Point(0,0));
        }
        
        public function getState():int {
            return this.state;
        }
    }
}

package UIComponents
{
    import flash.display.Bitmap;
    import flash.display.BitmapData;
    import flash.display.Sprite;
    import flash.events.*;
    import flash.geom.Matrix;
    import flash.geom.Point;
    import flash.geom.Rectangle;
            
    public class Map extends Sprite
    {
        private var tilePos:Array = 
        [
            {type:"blank", row:0, col:3},
            {type:"1", row:1, col:0},
            {type:"2", row:1, col:1},
            {type:"3", row:1, col:2},
            {type:"4", row:1, col:3},
            {type:"5", row:2, col:0},
            {type:"6", row:2, col:1},
            {type:"7", row:2, col:2},
            {type:"8", row:2, col:3},
            {type:"mine", row:0, col:2},
            {type:"flag", row:0, col:1},                
            {type:"unexplored", row:0, col:0}
        ];
        
        public const tileSize:int = 16;
        private var tiles:Bitmap;
        private var mapBMD:BitmapData;
        
        public function Map(resource:Bitmap, mapArr:Array, rowNum:int, colNum:int)
        {
            //scale resource image to a suitable size
               tiles = new Bitmap();
            tiles.bitmapData = new BitmapData(tileSize*4, tileSize*3, true);
            var scaleMat:Matrix = new Matrix();
            scaleMat.scale(tileSize*4/resource.width, tileSize*3/resource.height);
            tiles.bitmapData.draw(resource, scaleMat);    
            
            //initial state is full unexplored tiles
            mapBMD = new BitmapData(tileSize * colNum, tileSize * rowNum, false, 0xFFFFFF);
            var map:Bitmap = new Bitmap(mapBMD);
            var rect:Rectangle = new Rectangle(0,0,tileSize, tileSize);  
            for(var i:int=0;i<rowNum;i++)
                for(var j:int=0;j<colNum;j++)                
                     mapBMD.copyPixels(tiles.bitmapData, rect, new Point(j*tileSize,i*tileSize));                        

            addChild(map);
        }

        public function drawTile(i:int, j:int, type:int):void {
            var rect:Rectangle;
            if(type >= 0) //either blank or has a hint number
            rect = new Rectangle(tilePos[type].col * tileSize, tilePos[type].row * tileSize,tileSize, tileSize);
            else if (type < 0) //it is a mine
            rect = new Rectangle(tilePos[9].col * tileSize, tilePos[9].row * tileSize,tileSize, tileSize);
            
            mapBMD.copyPixels(tiles.bitmapData, rect, new Point(j * tileSize, i * tileSize));
        }

        
    }
}

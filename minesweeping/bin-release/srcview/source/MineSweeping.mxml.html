<?xml version="1.0" encoding="utf-8"?>
<mx:Application xmlns:mx="http://www.adobe.com/2006/mxml" 
    layout="absolute" width="400" height="300" backgroundColor="#b2f661"
     creationComplete="init()"
     themeColor="#B390FA" borderColor="#F6F161" viewSourceURL="srcview/index.html">
    
    <mx:Script>
        <![CDATA[
            import mx.rpc.events.FaultEvent;
            import mx.rpc.events.ResultEvent;
            import Game.MineSweeper;
            import mx.core.UIComponent;
            import mx.events.CloseEvent;
            import mx.controls.Alert;
            import flash.external.ExternalInterface;
            import MyLoader.MyImageLoader;
            
            private var game:MineSweeper;
            
            //resource urls
            private var urls:Array = [
                                  "resources/minesweeper_tiles.jpg", 
                                  "resources/smile.gif", 
                                  "resources/cool.png", 
                                  "resources/wrong.png", 
                                  "resources/digits_green.png",
                                  "resources/digits_red.png",
                                  "resources/digits_black.png"
                                  ];        
            
            private function init():void  {
                //instantiate an empty new game    and add it to stage    
                game = new MineSweeper();                
                window.width = this.width;
                window.height = this.height;
                var uic:UIComponent = new UIComponent();
                uic.x = 100;
                uic.addChild(game);
                window.addChild(uic);
                loadResources(); 
                
                //add keyboard event listners
                application.addEventListener(KeyboardEvent.KEY_DOWN, onKeyDown);
                application.addEventListener(KeyboardEvent.KEY_UP, onKeyUp);
                this.setFocus();
    
            }
            
            private function loadResources():void {
                    var myloader:MyImageLoader  = new MyImageLoader();
                    myloader.load(urls);
                    myloader.addEventListener("AllImagesLoaded", onImagesLoadingComplete);
            }
        
            private function onImagesLoadingComplete(e:Event):void {
                game.imgs = e.target as MyImageLoader; 
                game.urls = urls;   
                //upon images all loaded, request last saved data from database     
                game.startNewGame();                
                dbReadService.send(); 
            }            

            private function dbReadServiceResultHandler(e:ResultEvent):void {
                var result:Object = e.result;
                //
                if(result) {
                    try {
                        game.colNum = parseInt(result.state.cols);
                        game.rowNum = parseInt(result.state.rows);
                        game.mineNum = parseInt(result.state.mines);
                        game.mapArr = parseArray(result.state.mapArr, game.rowNum, game.colNum);
                        game.exploredArr = parseArray(result.state.exploredArr, game.rowNum, game.colNum);
                        game.timer.repeatCount = parseInt(result.state.time);
                        game.updateStatus();
                    } catch(err:Error) {
                        Alert.show("An error occurred while loading data. The game might not run.");
                    }
                } 
                
                //if result is null, just use the new game instance
                
            }
            
            private function dbWriteServiceResultHandler(e:ResultEvent):void {
                Alert.show("Your game data has been saved", "", 1, null, alertClickHandler);
                
            }

            
            private function savedata():void {
                var param:Object = {
                    rows: game.rowNum,
                    cols: game.colNum,
                    mines: game.mineNum,
                    time: game.timer.currentCount,
                    mapArr: array2String(game.mapArr),
                    exploredArr: array2String(game.exploredArr)
                }
                dbWriteService.send(param);                
            }
            
            
            private function alertClickHandler(e:Event):void {
                this.setFocus();
                ExternalInterface.call("closeBrowserWindow", null);
            }
            
            private function dbServiceFaultHandler(e:FaultEvent):void {
                Alert.show(e.message.toString());
            }
            
            
            private function parseArray(str:String, rows:int, cols:int):Array {
                    var arr:Array = new Array();
                    var temp:Array = str.split(","); //temp is an array of strings
                    for(var i:int = 0; i < rows; i++) {
                        arr[i] = new Array();                                            
                        for(var j:int = 0; j < cols; j++)
                            arr[i].push(int(temp[i*cols + j]));
                    }
                    return arr;        
            }
            
            
            private function array2String(arr:Array):String { 
                var str:String = "";
                for(var i:int = 0; i< arr.length; i++)
                    for(var j:int = 0; j<arr[i].length; j++)
                        if(i==0 && j==0) str += arr[i][j].toString();
                        else str += "," + arr[i][j].toString();
                return str;        
            }
            
                    
            private function onKeyDown(e:KeyboardEvent):void {
                if(e.keyCode == 32) game.flag = true; //the space key is pressed down
            }
    
            private function onKeyUp(e:KeyboardEvent):void {
                if(e.keyCode == 32) game.flag = false; //the space key is released
            }
            
        ]]>
    </mx:Script>
    
    
    <mx:Panel id="window" layout="absolute" borderAlpha="0.3"
        color="#000000"  borderColor="#FFFFFF" backgroundColor="#AA89F7">
        <mx:Text x="30" y="50" text = "Instructions" fontSize="16" color="#FFFFFF"/>
        <mx:Text x="30" y="100" width="150" color="#FFFFFF" text = "Add a flag: pressed space and click the tile at the same time"/>
        <mx:Text x="30" y="150" width="150" color="#FFFFFF" text = "Click 'Save' button to save the current game to database"/>
        <mx:Button x="30" y="200" label="Save" click="savedata()"/>
    </mx:Panel>
    
    <mx:HTTPService id = "dbWriteService" method="POST"
        url="php/dbSave.php"
        useProxy="false"
         resultFormat="text"
        result="dbWriteServiceResultHandler(event)"
        fault="dbServiceFaultHandler(event)">
    </mx:HTTPService>

    <mx:HTTPService id = "dbReadService" method="GET"
        url="php/dbRead.php"
        useProxy="false"
         resultFormat="object"
        result="dbReadServiceResultHandler(event)"
        fault="dbServiceFaultHandler(event)">
    </mx:HTTPService>
    
</mx:Application>
